167 int searchGrid::findCellIndex(double x,double y){
168  // find the raster indices of bucket containing x,y
169  int ix = static_cast<int>((x-x0) /_xSize * NxCells);
170  int iy = static_cast<int>((y-y0) /_ySize * NyCells);
171  //stop agents exactly on the extreme right boundary (or overflowing it) from wrapping around into the next row up
172  if (ix==NxCells)ix--;
173  //similarly make sure we don't overflow in y
174  if (iy==NyCells)iy--;
175  //make sure we are n't below zero
176  if (ix<0)ix=0;
177  if (iy<0)iy=0;
178  // return the bucket's linear index
179  return iy * NxCells + ix;
180 }

586 vector <place*> searchGrid::inRadius(agent* a,unsigned placeType, double d){
587  //find all places of a given type within a given distance of agent a
588  //may need to search multiple cells
589  //return results sorted by distance
590  vector<place*> temp;
591  vector<std::pair<place*,double>>pairsForSort;
592  int q=a->cellIndex;
593  int lx=q%NxCells; int ly=q/NxCells;
594  int nx=int(d/_xSize*NxCells+0.5);
595  int ny=int(d/_ySize*NyCells+0.5);
596  int il=-nx,iu=nx,jl=-nx,ju=nx;
597  //The domain is spanned! avoid duplications
598  if (nx>=NxCells/2) {il=-lx;iu=NxCells-1-lx;}
599  if (ny>=NyCells/2) {jl=-ly;ju=NyCells-1-ly;}
600  //hard edges
601  if (!cylX) {if (lx<nx) il=-lx;if (lx>=NxCells-1-nx) iu=NxCells-1-lx;}
602  if (!cylY) {if (ly<ny) jl=-ly;if (ly>=NyCells-1-ny) ju=NyCells-1-ly;}
603 
604  for (int i=il;i<iu+1;i++) {
605  for (int j=jl;j<ju+1;j++) {
606  //wrapping
607  int inx=lx+i; if (cylX) {while (inx<0)inx+=NxCells; while (inx>=NxCells)inx-=NxCells;}
608  int iny=ly+j; if (cylY) {while (iny<0)iny+=NyCells; while (iny>=NyCells)iny-=NyCells;}
609  int ind=inx+iny*NxCells;
610  if(ind>=0 && ind< (int)cells.size()) inDist(ind,d,a,placeType,pairsForSort);
611  }
612  }
613  //use lamba function for sort
614  std::sort(pairsForSort.begin(),pairsForSort.end(),
615  [](const std::pair<place*,double> &a, const std::pair<place*,double> &b) -> bool
616  {
617  return a.second < b.second;
618  });
619  for (auto& pr:pairsForSort)temp.push_back(pr.first);
620  return temp;
621 }
622 //------------------------------------------------------------------
623 //return all places of a given type in a cell within a given distance of agent a
624 void searchGrid::inDist(int cellIndex,float d,agent* a,unsigned placeType, vector<std::pair<place*,double>>& L){
625  auto range=locations[cellIndex].equal_range(placeType);
626  for (auto& it=range.first;it!=range.second;it++){
627  place* p=it->second;
628  double dx=(p->loc.x - a->loc.x);
629  double dy=(p->loc.y - a->loc.y);
630  //wrapping of distance measures
631  if (cylX && abs(dx)>_xSize/2)dx=_xSize-abs(dx);
632  if (cylY && abs(dy)>_ySize/2)dy=_ySize-abs(dy);
633  const double dist=dx*dx+dy*dy;
634  std::pair<place*,double> pear=std::make_pair(p,dist);
635  if (dist <= d*d)L.push_back(pear);
636  }
637 }
